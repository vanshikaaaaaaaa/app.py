import streamlit as st
from transformers import pipeline

st.title("🍽️ Restaurant Generator AI")

# Load model
pipe = pipeline("text-generation", model="gpt2")

# Input from user
cuisine = st.text_input("Enter Cuisine (e.g., Indian, Italian)")

if st.button("Generate"):
    if cuisine:
        prompt = f"""
        Suggest a restaurant name and 4 menu items for {cuisine} cuisine.
        
        Format:
        Restaurant name:
        Menu items:
        - item1
        - item2
        - item3
        - item4
        """

        result = pipe(prompt, max_new_tokens=60)
        output = result[0]['generated_text']

        st.subheader("✨ AI Suggestion")
        st.write(output)
    else:
        st.warning("Please enter a cuisine")
